5carddraw/ ├── server.js └── index.html const express = require("express"); const cors = require("cors"); const app = express(); app.use(cors()); app.use(express.json()); // ---------------- DATABASE ---------------- let users = {}; // ---------------- SUBMIT GAME ---------------- app.post("/submit", (req, res) => { const { username, total } = req.body; if (!users[username]) { users[username] = { username, today: 0, month: 0, lastPlayed: "" }; } let today = new Date().toDateString(); // ONE PLAY PER DAY RULE if (users[username].lastPlayed === today) { return res.json({ error: "Already played today" }); } users[username].today = total; users[username].month += total; users[username].lastPlayed = today; res.json({ success: true, total }); }); // ---------------- LEADERBOARDS ---------------- app.get("/leaderboard", (req, res) => { let sorted = Object.values(users).sort((a,b)=>b.today - a.today); res.json(sorted); }); app.get("/monthly", (req, res) => { let sorted = Object.values(users).sort((a,b)=>b.month - a.month); res.json(sorted); }); // ---------------- START SERVER ---------------- app.listen(3000, () => { console.log("🔥 5Card Draw server running on port 3000"); }); 5Card Draw
